52 const Maneuver &Planner::update(const point position, const double heading,
53  const double velocity,
54  const double dt, // TODO(roman): braucht es dt?
55  std::vector<polygon> obstacles) {
56  _maneuvers.clear();
57 
58  // add obstacles to spatial index
59  // TODO(roman): move to own function
60  for (auto obstacle : obstacles) {
61  box b = bg::return_envelope<box>(obstacle);
62  auto p = std::make_shared<polygon>(obstacle);
63  _rtree->insert(std::make_pair(b, p));
64  }
65 
66  std::pair<double, double> p =
67  _baseframe->localize(position, velocity, heading, dt, _last_known_s);
68  _last_known_s = p.first; // arc-length at current position
69  const double q_i = p.second; // lateral offset at current position
70 
71  for (double q_f = -_q_max; q_f <= _q_max; q_f += (2 * _q_max / _m_max)) {
72  Maneuver m = Maneuver(position, heading, velocity, q_i, q_f, _last_known_s,
73  _baseframe, _rtree);
74 
75  if (m.driveable()) { // only keep reasonable maneuvers
76  _maneuvers.push_back(m);
77  }
78  }
79 
80  // throw exception on this
81  CHECK(!_maneuvers.empty());
82 
83  std::map<double, Maneuver> candidates;
84  std::map<double, Maneuver> collisions_paths;
85 
86  // collision check
87  for (auto &m : _maneuvers) {
88  if (m.collision()) {
89  collisions_paths.insert(std::make_pair(m.collision(), m));
90  } else { // calculate path costs
91  double cost =
92  m.J(_maneuver, std::make_shared<std::vector<Maneuver>>(_maneuvers));
93  candidates.insert(std::make_pair(cost, m));
94  }
95  }
96 
97  _maneuver = candidates.empty() ? collisions_paths.rbegin()->second
98  : candidates.begin()->second;
99 
100  return _maneuver;
101 }
